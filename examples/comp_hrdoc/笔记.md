```shell
  cd /data/LLM_group/layoutlmft && \
  conda activate layoutxlm && \
  nohup python examples/comp_hrdoc/scripts/train_doc.py \
      --env test \
      --batch-size 1 \
      --use-semantic \
      --num-epochs 20 \
      > logs/train_doc_$(date +%Y%m%d_%H%M%S).log 2>&1 &


  参数说明

  | 参数            | 值       | 说明                         |
  |-----------------|----------|------------------------------|
  | --env test      | test     | 使用服务器环境配置           |
  | --batch-size 1  | 1        | 批次大小为1                  |
  | --use-construct | 默认开启 | 启用 Construct 模块 (4.3)    |
  | --use-semantic  | 需指定   | 启用 Semantic 分类模块 (4.2) |
  | --num-epochs    | 20       | 训练轮数                     |

  可选参数

  --model-type doc          # 模型类型：doc(完整) 或 order-only
  --hidden-size 768         # 隐藏层维度
  --order-num-layers 3      # Order Transformer 层数
  --construct-num-layers 3  # Construct Transformer 层数
  --learning-rate 5e-5      # 学习率
  --gradient-accumulation-steps 2  # 梯度累积（可弥补小batch）
  --fp16                    # 混合精度训练
  --quick                   # 快速测试模式
```
